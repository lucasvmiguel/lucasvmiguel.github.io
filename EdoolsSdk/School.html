def self.create(token, props)
  response = HTTP
    .headers('Authorization' => "Token token=\"#{token}\"")
    .post(SCHOOL_URL_CREATE, :json => props)

  return response if response.status != 201

  body = response.parse(:json)

  School.parse_json_created(body)
end
